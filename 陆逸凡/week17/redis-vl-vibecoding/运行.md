## 快速使用示例

### 1. SemanticCache — 缓存 LLM 问答



```python
import asyncio
from vecstore import *

async def main():
    cm = RedisConnectionManager(RedisConfig(url="redis://localhost:6379"))
    embedder = OpenAIEmbeddingProvider(model="text-embedding-3-small")

    cache = SemanticCache(
        connection_manager=cm,
        embedding_provider=embedder,
        config=SemanticCacheConfig(distance_threshold=0.5),  # [0, 2]
        session_id="user-123",
    )
    await cache.initialize_index()

    # 第一次：缓存未命中 → 调用 LLM 生成 → 存储
    answer = await cache.retrieve("法国的首都是哪里？")
    if answer is None:
        answer = "巴黎。"  # 这里替换为你的 LLM 调用
        await cache.store("法国的首都是哪里？", answer)
        print("MISS — 已缓存新答案")
    else:
        print("HIT — 缓存命中:", answer)

    # 第二次：语义相似的问法 → 缓存命中（不调 LLM）
    answer = await cache.retrieve("法国首都是？")
    print("缓存返回:", answer)  # => "巴黎。"

    await cm.close()

asyncio.run(main())
```

### 2. EmbeddingsCache — 避免重复 Embedding 计算



```python
cache = EmbeddingsCache(connection_manager=cm, embedding_provider=embedder)

# 第一次：计算并缓存
v1 = await cache.get_or_embed("什么是机器学习？")

# 第二次：直接返回缓存（不调用 API）
v2 = await cache.get_or_embed("什么是机器学习？")

# 批量模式
vectors = await cache.get_or_embed_many(["你好", "世界", "AI"])
```

### 3. SemanticMessageHistory — 带语义理解的聊天记录



```python
history = SemanticMessageHistory(
    connection_manager=cm,
    embedding_provider=embedder,
    session_id="对话-42",
)
await history.initialize_index()

await history.add_message("user", "法国的首都是哪里？")
await history.add_message("assistant", "巴黎。")
await history.add_message("user", "德国呢？")

# 语义检索 — 找到意思相近的历史记录，不论位置
results = await history.search_similar("给我讲讲法国")
for msg in results:
    print(f"[{msg.role}] {msg.content}")

# 获取最近消息（按时间）
recent = await history.get_recent(5)
```

### 4. SemanticRouter — 意图识别



```python
routes = [
    Route(name="greeting", examples=["你好", "嗨", "早上好"],
           description="问候"),
    Route(name="weather", examples=["天气怎么样", "今天下雨吗"],
           description="查天气"),
]

router = SemanticRouter(
    connection_manager=cm,
    embedding_provider=embedder,
    routes=routes,
)
await router.initialize()

match = await router.route("早上好啊！")
if match:
    print(f"意图: {match.route.name} (置信度: {match.confidence:.2f})")
    # 可以调用 match.route.handler(...) 分发处理
```

### 配置方式

可通过环境变量（`VECSTORE_` 前缀）或 `.env` 文件配置：



```bash
# .env 文件
VECSTORE_REDIS_URL=redis://localhost:6379
VECSTORE_EMBEDDING_PROVIDER=openai
VECSTORE_EMBEDDING_MODEL=text-embedding-3-small
VECSTORE_SEMANTIC_CACHE_THRESHOLD=0.5
```

也可以在代码中直接传参创建各组件
