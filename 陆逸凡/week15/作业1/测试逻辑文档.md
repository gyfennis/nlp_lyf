## 五、评估脚本

### 5.1 综合评估器实现

```python
# eval/evaluator.py

import re
from typing import Dict, List, Tuple

class RAGEvaluator:
    """RAG系统评估器"""
    
    def __init__(self, test_queries: List[Dict], client):
        self.test_queries = test_queries
        self.client = client
    
    def evaluate_all(self) -> Dict:
        """执行全部评估"""
        results = []
        for query in self.test_queries:
            result = self.evaluate_single(query)
            results.append(result)
        
        return {
            "overall_score": sum(r["total_score"] for r in results) / len(results),
            "detailed_results": results,
            "metrics": self.calculate_metrics(results)
        }
    
    def evaluate_single(self, query: Dict) -> Dict:
        """评估单个查询"""
        # 调用RAG接口
        response = self.client.post("/chat", json={
            "question": query["question"],
            "knowledge_base_id": "test_kb"
        })
        result = response.json()
        
        # 计算三个指标
        filename_score = self.calc_filename_match(
            result["sources"], 
            query["expected_source"]["filename"]
        )
        page_score = self.calc_page_match(
            result["sources"], 
            query["expected_source"]["page"]
        )
        content_score = self.calc_content_similarity(
            result["answer"], 
            query["expected_answer"]
        )
        
        total_score = filename_score * 0.25 + page_score * 0.25 + content_score * 0.5
        
        return {
            "query_id": query["id"],
            "filename_score": filename_score,
            "page_score": page_score,
            "content_score": content_score,
            "total_score": total_score,
            "answer": result["answer"],
            "sources": result["sources"]
        }
    
    def calc_filename_match(self, sources: List[Dict], expected_filename: str) -> float:
        """计算文件名匹配度（0-1）"""
        for source in sources:
            if expected_filename in source.get("filename", ""):
                return 1.0
        return 0.0
    
    def calc_page_match(self, sources: List[Dict], expected_page: int) -> float:
        """计算页面匹配度（0-1）"""
        for source in sources:
            if source.get("page") == expected_page:
                return 1.0
        return 0.0
    
    def calc_content_similarity(self, generated: str, expected: str) -> float:
        """计算Jaccard相似系数"""
        # 分词（简单处理，可替换为jieba）
        def tokenize(text: str) -> set:
            # 提取中文和英文单词
            chinese = re.findall(r'[\u4e00-\u9fff]+', text)
            english = re.findall(r'[a-zA-Z0-9]+', text)
            return set(chinese + english)
        
        gen_tokens = tokenize(generated.lower())
        exp_tokens = tokenize(expected.lower())
        
        if not exp_tokens:
            return 1.0 if not gen_tokens else 0.0
        
        intersection = len(gen_tokens & exp_tokens)
        union = len(gen_tokens | exp_tokens)
        
        return intersection / union if union > 0 else 0.0
    
    def calculate_metrics(self, results: List[Dict]) -> Dict:
        """计算汇总指标"""
        total = len(results)
        return {
            "avg_filename_score": sum(r["filename_score"] for r in results) / total,
            "avg_page_score": sum(r["page_score"] for r in results) / total,
            "avg_content_score": sum(r["content_score"] for r in results) / total,
            "avg_total_score": sum(r["total_score"] for r in results) / total,
            "perfect_scores": sum(1 for r in results if r["total_score"] == 1.0),
            "failed_scores": sum(1 for r in results if r["total_score"] < 0.5)
        }

# 运行评估
if __name__ == "__main__":
    evaluator = RAGEvaluator(test_queries, client)
    results = evaluator.evaluate_all()
    
    print(f"Overall Score: {results['overall_score']:.3f}")
    print(f"Metrics: {results['metrics']}")
    
    # 生成HTML报告
    generate_html_report(results, "evaluation_report.html")

        <h2>失败案例分析</h2>
        {generate_failure_analysis(results['detailed_results'])}
    </body>
    </html>
    """
    
    with open(output_path, 'w', encoding='utf-8') as f:
        f.write(html)
```
